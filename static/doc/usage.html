Icons - top

This template have 4 different icon library. Wrap all the icon with the i tag. Below is the list of available icon

  1. Glyph Icon - Bootstrap default
  2. Elusive Icon
  3. Fugue Icon
  4. Font Awesome Icon

For an example see the below code.

<i class="icon icon-cog"></i> // Glyph icon
<i class="icon iconfa-cog"></i> // Font Awesome icon
<i class="icon iconelusive-cog"></i> // Elusive icon
<i class="icon iconfugue16-balloon"></i> // Fugue icon

Example HTML Code

<div id="calendar"></div>

Example Javascript Code

$(function() {
	var date = new Date();
	var d = date.getDate();
	var m = date.getMonth();
	var y = date.getFullYear();
	
	$('#calendar').fullCalendar({
		theme: true,
		header: {
			left: 'prev,next today',
			center: 'title',
			right: 'month,agendaWeek,agendaDay'
		},
		editable: true,
		events: [
			{
				title: 'Long Event',
				start: new Date(y, m, d-5),
				end: new Date(y, m, d-2)
			}
		]
	});
});

Javascript File

<!-- Plugins -->
<script src="js/plugins/jquery.shuffle.min.js"></script>

Sample Script

$(function() {
	// Button Type Filter
	$('#filter-button button').on('click', function() {
		var $this = $(this),
			$gallery = $('#gallery');

		// Filter elements
		$gallery.shuffle($this.data('group'));
	});
	
	// Instantiate the plugin
	$('#gallery').shuffle({
		group : 'all',
		speed : 800,
		easing : 'ease-out'
	});
});

Usage Example - The Javascript

$(function () {
   var data = [{
       label : 'Population',
       color : '#B74231',
       shadowSize : 0,
       data : [["Jan", 450],["Feb", 550],["Mar", 320],["Apr", 700],["May", 200],["Jun", 330],["Jul", 900],["Aug", 140],["Sep", 300],["Nov", 500],["Dec", 300]]
   }]
   
   $.plot($("#chart1"), data, {
       series: {
           bars: {
               show: true,
               barWidth: 0.3,
               align: "center" ,
               fill: .8
           }
       },
       grid: {
           borderWidth: 0, 
           hoverable: true 
       },
       tooltip: true,
       tooltipOpts: {
           content: "%x : %y"
       },
       xaxis: {
           mode: "categories",
           tickLength: 0
       }
   });
});
